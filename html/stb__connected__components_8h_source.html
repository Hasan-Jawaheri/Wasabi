1 // stb_connected_components - v0.94 - public domain connected components on grids
2 // http://github.com/nothings/stb
3 //
4 // Finds connected components on 2D grids for testing reachability between
5 // two points, with fast updates when changing reachability (e.g. on one machine
6 // it was typically 0.2ms w/ 1024x1024 grid). Each grid square must be "open" or
7 // "closed" (traversable or untraversable), and grid squares are only connected
8 // to their orthogonal neighbors, not diagonally.
9 //
10 // In one source file, create the implementation by doing something like this:
11 //
12 // #define STBCC_GRID_COUNT_X_LOG2 10
13 // #define STBCC_GRID_COUNT_Y_LOG2 10
14 // #define STB_CONNECTED_COMPONENTS_IMPLEMENTATION
15 // #include "stb_connected_components.h"
16 //
17 // The above creates an implementation that can run on maps up to 1024x1024.
18 // Map sizes must be a multiple of (1<<(LOG2/2)) on each axis (e.g. 32 if LOG2=10,
19 // 16 if LOG2=8, etc.) (You can just pad your map with untraversable space.)
20 //
21 // MEMORY USAGE
22 //
23 // Uses about 6-7 bytes per grid square (e.g. 7MB for a 1024x1024 grid).
24 // Uses a single worst-case allocation which you pass in.
25 //
26 // PERFORMANCE
27 //
28 // On a core i7-2700K at 3.5 Ghz, for a particular 1024x1024 map (map_03.png):
29 //
30 // Creating map : 44.85 ms
31 // Making one square traversable: 0.27 ms (average over 29,448 calls)
32 // Making one square untraversable: 0.23 ms (average over 30,123 calls)
33 // Reachability query: 0.00001 ms (average over 4,000,000 calls)
34 //
35 // On non-degenerate maps update time is O(N^0.5), but on degenerate maps like
36 // checkerboards or 50% random, update time is O(N^0.75) (~2ms on above machine).
37 //
38 // CHANGELOG
39 //
40 // 0.94 (2016-04-17) Bugfix & optimize worst case (checkerboard & random)
41 // 0.93 (2016-04-16) Reduce memory by 10x for 1Kx1K map; small speedup

58 // ALGORITHM
59 //
60 // The NxN grid map is split into sqrt(N) x sqrt(N) blocks called
61 // "clusters". Each cluster independently computes a set of connected
62 // components within that cluster (ignoring all connectivity out of
63 // that cluster) using a union-find disjoint set forest. This produces a bunch
64 // of locally connected components called "clumps". Each clump is (a) connected
65 // within its cluster, (b) does not directly connect to any other clumps in the
66 // cluster (though it may connect to them by paths that lead outside the cluster,
67 // but those are ignored at this step), and (c) maintains an adjacency list of
68 // all clumps in adjacent clusters that it _is_ connected to. Then a second
69 // union-find disjoint set forest is used to compute connected clumps
70 // globally, across the whole map. Reachability is then computed by
71 // finding which clump each input point belongs to, and checking whether
72 // those clumps are in the same "global" connected component.
73 //
74 // The above data structure can be updated efficiently; on a change
75 // of a single grid square on the map, only one cluster changes its
76 // purely-local state, so only one cluster needs its clumps fully
77 // recomputed. Clumps in adjacent clusters need their adjacency lists
78 // updated: first to remove all references to the old clumps in the
79 // rebuilt cluster, then to add new references to the new clumps. Both
80 // of these operations can use the existing "find which clump each input
81 // point belongs to" query to compute that adjacency information rapidly.

270 int stbcc_query_grid_node_connection(stbcc_grid *g, int x1, int y1, int x2, int y2)
271 {
272  stbcc__global_clumpid label1, label2;
273  stbcc__clumpid c1 = g->clump_for_node[y1][x1];
274  stbcc__clumpid c2 = g->clump_for_node[y2][x2];
275  int cx1 = STBCC__CLUSTER_X_FOR_COORD_X(x1);
276  int cy1 = STBCC__CLUSTER_Y_FOR_COORD_Y(y1);
277  int cx2 = STBCC__CLUSTER_X_FOR_COORD_X(x2);
278  int cy2 = STBCC__CLUSTER_Y_FOR_COORD_Y(y2);
279  assert(!g->in_batched_update);
280  if (c1 == STBCC__NULL_CLUMPID || c2 == STBCC__NULL_CLUMPID)
281  return 0;
282  label1 = g->cluster[cy1][cx1].clump[c1].global_label;
283  label2 = g->cluster[cy2][cx2].clump[c2].global_label;
284  if (label1.c == label2.c)
285  return 1;
286  return 0;
287 }

352 static void stbcc__build_connected_components_for_clumps(stbcc_grid *g)
353 {
354  int i,j,k,h;
355 
356  for (j=0; j < STBCC__CLUSTER_COUNT_Y; ++j) {
357  for (i=0; i < STBCC__CLUSTER_COUNT_X; ++i) {
358  stbcc__cluster *cluster = &g->cluster[j][i];
359  for (k=0; k < (int) cluster->num_edge_clumps; ++k) {
360  stbcc__global_clumpid m;
361  m.f.clump_index = k;
362  m.f.cluster_x = i;
363  m.f.cluster_y = j;
364  assert((int) m.f.clump_index == k && (int) m.f.cluster_x == i && (int) m.f.cluster_y == j);
365  cluster->clump[k].global_label = m;
366  }
367  }
368  }
369 
370  for (j=0; j < STBCC__CLUSTER_COUNT_Y; ++j) {
371  for (i=0; i < STBCC__CLUSTER_COUNT_X; ++i) {
372  stbcc__cluster *cluster = &g->cluster[j][i];
373  for (k=0; k < (int) cluster->num_edge_clumps; ++k) {
374  stbcc__clump *clump = &cluster->clump[k];
375  stbcc__unpacked_clumpid m;
376  stbcc__relative_clumpid *adj;
377  m.clump_index = k;
378  m.cluster_x = i;
379  m.cluster_y = j;
380  adj = &cluster->adjacency_storage[clump->adjacent_clump_list_index];
381  for (h=0; h < clump->num_adjacent; ++h) {
382  unsigned int clump_index = adj[h].clump_index;
383  unsigned int x = adj[h].cluster_dx + i;
384  unsigned int y = adj[h].cluster_dy + j;
385  stbcc__clump_union(g, m, x, y, clump_index);
386  }
387  }
388  }
389  }
390 
391  for (j=0; j < STBCC__CLUSTER_COUNT_Y; ++j) {
392  for (i=0; i < STBCC__CLUSTER_COUNT_X; ++i) {
393  stbcc__cluster *cluster = &g->cluster[j][i];
394  for (k=0; k < (int) cluster->num_edge_clumps; ++k) {
395  stbcc__global_clumpid m;
396  m.f.clump_index = k;
397  m.f.cluster_x = i;
398  m.f.cluster_y = j;
399  stbcc__clump_find(g, m);
400  }
401  }
402  }

405 static void stbcc__build_all_connections_for_cluster(stbcc_grid *g, int cx, int cy)
406 {
407  // in this particular case, we are fully non-incremental. that means we
408  // can discover the correct sizes for the arrays, but requires we build
409  // the data into temporary data structures, or just count the sizes, so
410  // for simplicity we do the latter
411  stbcc__cluster *cluster = &g->cluster[cy][cx];
412  unsigned char connected[STBCC__MAX_CLUMPS_PER_CLUSTER/8];
413  unsigned char num_adj[STBCC__MAX_CLUMPS_PER_CLUSTER] = { 0 };
414  int x = cx * STBCC__CLUSTER_SIZE_X;
415  int y = cy * STBCC__CLUSTER_SIZE_Y;
416  int step_x, step_y=0, i, j, k, n, m, dx, dy, total;
417  int extra;
418 
419  g->cluster[cy][cx].rebuild_adjacency = 0;
420 
421  total = 0;
422  for (m=0; m < 4; ++m) {
423  switch (m) {
424  case 0:
425  dx = 1, dy = 0;
426  step_x = 0, step_y = 1;
427  i = STBCC__CLUSTER_SIZE_X-1;
428  j = 0;
429  n = STBCC__CLUSTER_SIZE_Y;
430  break;
431  case 1:
432  dx = -1, dy = 0;
433  i = 0;
434  j = 0;
435  step_x = 0;
436  step_y = 1;
437  n = STBCC__CLUSTER_SIZE_Y;
438  break;
439  case 2:
440  dy = -1, dx = 0;
441  i = 0;
442  j = 0;
443  step_x = 1;
444  step_y = 0;
445  n = STBCC__CLUSTER_SIZE_X;
446  break;
447  case 3:
448  dy = 1, dx = 0;
449  i = 0;
450  j = STBCC__CLUSTER_SIZE_Y-1;
451  step_x = 1;
452  step_y = 0;
453  n = STBCC__CLUSTER_SIZE_X;
454  break;
455  }
456 
457  if (cx+dx < 0 || cx+dx >= g->cw || cy+dy < 0 || cy+dy >= g->ch)
458  continue;
459 
460  memset(connected, 0, sizeof(connected));
461  for (k=0; k < n; ++k) {
462  if (STBCC__MAP_OPEN(g, x+i, y+j) && STBCC__MAP_OPEN(g, x+i+dx, y+j+dy)) {
463  stbcc__clumpid c = g->clump_for_node[y+j+dy][x+i+dx];
464  if (0 == (connected[c>>3] & (1 << (c & 7)))) {
465  connected[c>>3] |= 1 << (c & 7);
466  ++num_adj[g->clump_for_node[y+j][x+i]];
467  ++total;
468  }
469  }
470  i += step_x;
471  j += step_y;
472  }
473  }
474 
475  assert(total <= STBCC__CLUSTER_ADJACENCY_COUNT);
476 
477  // decide how to apportion unused adjacency slots; only clumps that lie
478  // on the edges of the cluster need adjacency slots, so divide them up
479  // evenly between those clumps
480 
481  // we want:
482  // extra = (STBCC__CLUSTER_ADJACENCY_COUNT - total) / cluster->num_edge_clumps;
483  // but we efficiently approximate this without a divide, because
484  // ignoring edge-vs-non-edge with 'num_adj[i]*2' was faster than
485  // 'num_adj[i]+extra' with the divide
486  if (total + (cluster->num_edge_clumps<<2) <= STBCC__CLUSTER_ADJACENCY_COUNT)
487  extra = 4;
488  else if (total + (cluster->num_edge_clumps<<1) <= STBCC__CLUSTER_ADJACENCY_COUNT)
489  extra = 2;
490  else if (total + (cluster->num_edge_clumps<<0) <= STBCC__CLUSTER_ADJACENCY_COUNT)
491  extra = 1;
492  else
493  extra = 0;
494 
495  total = 0;
496  for (i=0; i < (int) cluster->num_edge_clumps; ++i) {
497  int alloc = num_adj[i]+extra;
498  if (alloc > STBCC__MAX_EXITS_PER_CLUSTER)
499  alloc = STBCC__MAX_EXITS_PER_CLUSTER;
500  assert(total < 256); // must fit in byte
501  cluster->clump[i].adjacent_clump_list_index = (unsigned char) total;
502  cluster->clump[i].max_adjacent = alloc;

856 static void stbcc__build_clumps_for_cluster(stbcc_grid *g, int cx, int cy)
857 {
858  stbcc__cluster *c;
859  stbcc__cluster_build_info cbi;
860  int label=0;
861  int i,j;
862  int x = cx * STBCC__CLUSTER_SIZE_X;
863  int y = cy * STBCC__CLUSTER_SIZE_Y;
864 
865  // set initial disjoint set forest state
866  for (j=0; j < STBCC__CLUSTER_SIZE_Y; ++j) {
867  for (i=0; i < STBCC__CLUSTER_SIZE_X; ++i) {
868  cbi.parent[j][i].x = i;
869  cbi.parent[j][i].y = j;
870  }
871  }
872 
873  // join all sets that are connected
874  for (j=0; j < STBCC__CLUSTER_SIZE_Y; ++j) {
875  // check down only if not on bottom row
876  if (j < STBCC__CLUSTER_SIZE_Y-1)
877  for (i=0; i < STBCC__CLUSTER_SIZE_X; ++i)
878  if (STBCC__MAP_OPEN(g,x+i,y+j) && STBCC__MAP_OPEN(g,x+i ,y+j+1))
879  stbcc__incluster_union(&cbi, i,j, i,j+1);
880  // check right for everything but rightmost column
881  for (i=0; i < STBCC__CLUSTER_SIZE_X-1; ++i)
882  if (STBCC__MAP_OPEN(g,x+i,y+j) && STBCC__MAP_OPEN(g,x+i+1,y+j ))
883  stbcc__incluster_union(&cbi, i,j, i+1,j);
884  }
885 
886  // label all non-empty clumps along edges so that all edge clumps are first
887  // in list; this means in degenerate case we can skip traversing non-edge clumps.
888  // because in the first pass we only label leaders, we swap the leader to the
889  // edge first
890 
891  // first put solid labels on all the edges; these will get overwritten if they're open
892  for (j=0; j < STBCC__CLUSTER_SIZE_Y; ++j)
893  cbi.label[j][0] = cbi.label[j][STBCC__CLUSTER_SIZE_X-1] = STBCC__NULL_CLUMPID;
894  for (i=0; i < STBCC__CLUSTER_SIZE_X; ++i)
895  cbi.label[0][i] = cbi.label[STBCC__CLUSTER_SIZE_Y-1][i] = STBCC__NULL_CLUMPID;
896 
897  for (j=0; j < STBCC__CLUSTER_SIZE_Y; ++j) {
898  i = 0;
899  if (STBCC__MAP_OPEN(g, x+i, y+j)) {
900  stbcc__tinypoint p = stbcc__incluster_find(&cbi, i,j);
901  if (p.x == i && p.y == j)
902  // if this is the leader, give it a label
903  cbi.label[j][i] = label++;
904  else if (!(p.x == 0 || p.x == STBCC__CLUSTER_SIZE_X-1 || p.y == 0 || p.y == STBCC__CLUSTER_SIZE_Y-1)) {
905  // if leader is in interior, promote this edge node to leader and label
906  stbcc__switch_root(&cbi, i, j, p);
907  cbi.label[j][i] = label++;
908  }
909  // else if leader is on edge, do nothing (it'll get labelled when we reach it)
910  }
911  i = STBCC__CLUSTER_SIZE_X-1;
912  if (STBCC__MAP_OPEN(g, x+i, y+j)) {
913  stbcc__tinypoint p = stbcc__incluster_find(&cbi, i,j);
914  if (p.x == i && p.y == j)
915  cbi.label[j][i] = label++;
916  else if (!(p.x == 0 || p.x == STBCC__CLUSTER_SIZE_X-1 || p.y == 0 || p.y == STBCC__CLUSTER_SIZE_Y-1)) {
917  stbcc__switch_root(&cbi, i, j, p);
918  cbi.label[j][i] = label++;
919  }
920  }
921  }
922 
923  for (i=1; i < STBCC__CLUSTER_SIZE_Y-1; ++i) {
924  j = 0;
925  if (STBCC__MAP_OPEN(g, x+i, y+j)) {
926  stbcc__tinypoint p = stbcc__incluster_find(&cbi, i,j);
927  if (p.x == i && p.y == j)
928  cbi.label[j][i] = label++;
929  else if (!(p.x == 0 || p.x == STBCC__CLUSTER_SIZE_X-1 || p.y == 0 || p.y == STBCC__CLUSTER_SIZE_Y-1)) {
930  stbcc__switch_root(&cbi, i, j, p);
931  cbi.label[j][i] = label++;
932  }
933  }
934  j = STBCC__CLUSTER_SIZE_Y-1;
935  if (STBCC__MAP_OPEN(g, x+i, y+j)) {
936  stbcc__tinypoint p = stbcc__incluster_find(&cbi, i,j);
937  if (p.x == i && p.y == j)
938  cbi.label[j][i] = label++;
939  else if (!(p.x == 0 || p.x == STBCC__CLUSTER_SIZE_X-1 || p.y == 0 || p.y == STBCC__CLUSTER_SIZE_Y-1)) {
940  stbcc__switch_root(&cbi, i, j, p);
941  cbi.label[j][i] = label++;
942  }
943  }
944  }

946  c = &g->cluster[cy][cx];
947  c->num_edge_clumps = label;
948 
949  // label any internal clusters
950  for (j=1; j < STBCC__CLUSTER_SIZE_Y-1; ++j) {
951  for (i=1; i < STBCC__CLUSTER_SIZE_X-1; ++i) {
952  stbcc__tinypoint p = cbi.parent[j][i];
953  if (p.x == i && p.y == j)
954  if (STBCC__MAP_OPEN(g,x+i,y+j))
955  cbi.label[j][i] = label++;
956  else
957  cbi.label[j][i] = STBCC__NULL_CLUMPID;
958  }
959  }
960 
961  // label all other nodes
962  for (j=0; j < STBCC__CLUSTER_SIZE_Y; ++j) {
963  for (i=0; i < STBCC__CLUSTER_SIZE_X; ++i) {
964  stbcc__tinypoint p = stbcc__incluster_find(&cbi, i,j);
965  if (p.x != i || p.y != j) {
966  if (STBCC__MAP_OPEN(g,x+i,y+j))
967  cbi.label[j][i] = cbi.label[p.y][p.x];
968  }
969  if (STBCC__MAP_OPEN(g,x+i,y+j))
970  assert(cbi.label[j][i] != STBCC__NULL_CLUMPID);
971  }
972  }
973

975

977  c->clump[i].num_adjacent = 0;
978  c->clump[i].max_adjacent = 0;
979  }
980 
981  for (j=0; j < STBCC__CLUSTER_SIZE_Y; ++j)
982  for (i=0; i < STBCC__CLUSTER_SIZE_X; ++i) {
983  g->clump_for_node[y+j][x+i] = cbi.label[j][i]; // @OPTIMIZE: remove cbi.label entirely
984  assert(g->clump_for_node[y+j][x+i] <= STBCC__NULL_CLUMPID);
985  }
986 
987  // set the global label for all interior clumps since they can't have connections,
988  // so we don't have to do this on the global pass (brings from O(N) to O(N^0.75))
989  for (i=(int) c->num_edge_clumps; i < (int) c->num_clumps; ++i) {
990  stbcc__global_clumpid gc;
991  gc.f.cluster_x = cx;
992  gc.f.cluster_y = cy;
993  gc.f.clump_index = i;
994  c->clump[i].global_label = gc;
