120  // Get available queue family properties
121  uint32_t queueCount;
122  vkGetPhysicalDeviceQueueFamilyProperties(physicalDevice, &queueCount, NULL);
123  assert(queueCount >= 1);
124 
125  std::vector<VkQueueFamilyProperties> queueProps(queueCount);
126  vkGetPhysicalDeviceQueueFamilyProperties(physicalDevice, &queueCount, queueProps.data());

128  // Iterate over each queue to learn whether it supports presenting:
129  // Find a queue with present support
130  // Will be used to present the swap chain images to the windowing system
131  std::vector<VkBool32> supportsPresent(queueCount);
132  for (uint32_t i = 0; i < queueCount; i++)
133  {
134  fpGetPhysicalDeviceSurfaceSupportKHR(physicalDevice, i, surface, &supportsPresent[i]);
135  }
136 
137  // Search for a graphics and a present queue in the array of queue
138  // families, try to find one that supports both
139  uint32_t graphicsQueueNodeIndex = UINT32_MAX;
140  uint32_t presentQueueNodeIndex = UINT32_MAX;
141  for (uint32_t i = 0; i < queueCount; i++)
142  {
143  if ((queueProps[i].queueFlags & VK_QUEUE_GRAPHICS_BIT) != 0)
144  {
145  if (graphicsQueueNodeIndex == UINT32_MAX)
146  {
147  graphicsQueueNodeIndex = i;
148  }
149 
150  if (supportsPresent[i] == VK_TRUE)
151  {
152  graphicsQueueNodeIndex = i;
153  presentQueueNodeIndex = i;
154  break;
155  }
156  }
157  }
158  if (presentQueueNodeIndex == UINT32_MAX)
159  {
160  // If there's no queue that supports both present and graphics
161  // try to find a separate present queue
162  for (uint32_t i = 0; i < queueCount; ++i)
163  {
164  if (supportsPresent[i] == VK_TRUE)
165  {
166  presentQueueNodeIndex = i;
167  break;
168  }
169  }
170  }
171 
172  // Exit if either a graphics or a presenting queue hasn't been found
173  if (graphicsQueueNodeIndex == UINT32_MAX || presentQueueNodeIndex == UINT32_MAX)
174  return false;
175 
176  // todo : Add support for separate graphics and presenting queue
177  if (graphicsQueueNodeIndex != presentQueueNodeIndex)
178  return false;
179 
180  queueNodeIndex = graphicsQueueNodeIndex;
181 
182  // Get list of supported surface formats
183  uint32_t formatCount;
184  err = fpGetPhysicalDeviceSurfaceFormatsKHR(physicalDevice, surface, &formatCount, NULL);
185  assert(!err);
186  assert(formatCount > 0);
187 
188  std::vector<VkSurfaceFormatKHR> surfaceFormats(formatCount);
189  err = fpGetPhysicalDeviceSurfaceFormatsKHR(physicalDevice, surface, &formatCount, surfaceFormats.data());
190  assert(!err);
191 
192  // If the surface format list only includes one entry with VK_FORMAT_UNDEFINED,
193  // there is no preferered format, so we assume VK_FORMAT_B8G8R8A8_UNORM
194  if ((formatCount == 1) && (surfaceFormats[0].format == VK_FORMAT_UNDEFINED))
195  {
196  colorFormat = VK_FORMAT_B8G8R8A8_UNORM;
197  }
198  else
199  {
200  // Always select the first available color format
201  // If you need a specific format (e.g. SRGB) you'd need to
202  // iterate over the list of available surface format and
203  // check for it's presence
204  colorFormat = surfaceFormats[0].format;
205  }
206  colorSpace = surfaceFormats[0].colorSpace;
207 
208  return true;

228  // Create the swap chain and get images with given width and height
229  void create(VkCommandBuffer cmdBuffer, uint32_t *width, uint32_t *height)
230  {
231  VkResult err;
232  VkSwapchainKHR oldSwapchain = swapChain;
233 
234  // Get physical device surface properties and formats
235  VkSurfaceCapabilitiesKHR surfCaps;
236  err = fpGetPhysicalDeviceSurfaceCapabilitiesKHR(physicalDevice, surface, &surfCaps);
237  assert(!err);
238 
239  // Get available present modes
240  uint32_t presentModeCount;
241  err = fpGetPhysicalDeviceSurfacePresentModesKHR(physicalDevice, surface, &presentModeCount, NULL);
242  assert(!err);
243  assert(presentModeCount > 0);
244 
245  std::vector<VkPresentModeKHR> presentModes(presentModeCount);
246 
247  err = fpGetPhysicalDeviceSurfacePresentModesKHR(physicalDevice, surface, &presentModeCount, presentModes.data());
248  assert(!err);
249 
250  VkExtent2D swapchainExtent = {};
251  // width and height are either both -1, or both not -1.
252  if (surfCaps.currentExtent.width == -1)
253  {
254  // If the surface size is undefined, the size is set to
255  // the size of the images requested.
256  swapchainExtent.width = *width;
257  swapchainExtent.height = *height;
258  }
259  else
260  {
261  // If the surface size is defined, the swap chain size must match
262  swapchainExtent = surfCaps.currentExtent;
263  *width = surfCaps.currentExtent.width;
264  *height = surfCaps.currentExtent.height;
265  }
266 
267  // Prefer mailbox mode if present, it's the lowest latency non-tearing present mode
268  VkPresentModeKHR swapchainPresentMode = VK_PRESENT_MODE_FIFO_KHR;
269  for (size_t i = 0; i < presentModeCount; i++)
270  {
271  if (presentModes[i] == VK_PRESENT_MODE_MAILBOX_KHR)
272  {
273  swapchainPresentMode = VK_PRESENT_MODE_MAILBOX_KHR;
274  break;
275  }
276  if ((swapchainPresentMode != VK_PRESENT_MODE_MAILBOX_KHR) && (presentModes[i] == VK_PRESENT_MODE_IMMEDIATE_KHR))
277  {
278  swapchainPresentMode = VK_PRESENT_MODE_IMMEDIATE_KHR;
279  }
280  }
281 
282  // Determine the number of images
283  uint32_t desiredNumberOfSwapchainImages = surfCaps.minImageCount + 1;
284  if ((surfCaps.maxImageCount > 0) && (desiredNumberOfSwapchainImages > surfCaps.maxImageCount))
285  {
286  desiredNumberOfSwapchainImages = surfCaps.maxImageCount;
287  }
288 
289  VkSurfaceTransformFlagsKHR preTransform;
290  if (surfCaps.supportedTransforms & VK_SURFACE_TRANSFORM_IDENTITY_BIT_KHR)
291  {
292  preTransform = VK_SURFACE_TRANSFORM_IDENTITY_BIT_KHR;
293  }
294  else
295  {
296  preTransform = surfCaps.currentTransform;
297  }
298 
299  VkSwapchainCreateInfoKHR swapchainCI = {};
300  swapchainCI.sType = VK_STRUCTURE_TYPE_SWAPCHAIN_CREATE_INFO_KHR;
301  swapchainCI.pNext = NULL;
302  swapchainCI.surface = surface;
303  swapchainCI.minImageCount = desiredNumberOfSwapchainImages;
304  swapchainCI.imageFormat = colorFormat;
305  swapchainCI.imageColorSpace = colorSpace;
306  swapchainCI.imageExtent = { swapchainExtent.width, swapchainExtent.height };
307  swapchainCI.imageUsage = VK_IMAGE_USAGE_COLOR_ATTACHMENT_BIT;
308  swapchainCI.preTransform = (VkSurfaceTransformFlagBitsKHR)preTransform;
309  swapchainCI.imageArrayLayers = 1;
310  swapchainCI.imageSharingMode = VK_SHARING_MODE_EXCLUSIVE;
311  swapchainCI.queueFamilyIndexCount = 0;
312  swapchainCI.pQueueFamilyIndices = NULL;
313  swapchainCI.presentMode = swapchainPresentMode;
314  swapchainCI.oldSwapchain = oldSwapchain;
315  swapchainCI.clipped = true;
316  swapchainCI.compositeAlpha = VK_COMPOSITE_ALPHA_OPAQUE_BIT_KHR;
317 
318  err = fpCreateSwapchainKHR(device, &swapchainCI, nullptr, &swapChain);
319  assert(!err);
320 
321  // If an existing sawp chain is re-created, destroy the old swap chain
322  // This also cleans up all the presentable images
323  if (oldSwapchain != VK_NULL_HANDLE)
324  {
325  for (uint32_t i = 0; i < imageCount; i++)
326  {
327  vkDestroyImageView(device, buffers[i].view, nullptr);
328  }
329  fpDestroySwapchainKHR(device, oldSwapchain, nullptr);
330  }
331 
332  err = fpGetSwapchainImagesKHR(device, swapChain, &imageCount, NULL);
333  assert(!err);
334 
335  // Get the swap chain images
336  images.resize(imageCount);
337  err = fpGetSwapchainImagesKHR(device, swapChain, &imageCount, images.data());
338  assert(!err);
339 
340  // Get the swap chain buffers containing the image and imageview
341  buffers.resize(imageCount);
342  for (uint32_t i = 0; i < imageCount; i++)
343  {
344  VkImageViewCreateInfo colorAttachmentView = {};
345  colorAttachmentView.sType = VK_STRUCTURE_TYPE_IMAGE_VIEW_CREATE_INFO;
346  colorAttachmentView.pNext = NULL;
347  colorAttachmentView.format = colorFormat;
348  colorAttachmentView.components = {
349  VK_COMPONENT_SWIZZLE_R,
350  VK_COMPONENT_SWIZZLE_G,
351  VK_COMPONENT_SWIZZLE_B,
352  VK_COMPONENT_SWIZZLE_A
353  };
354  colorAttachmentView.subresourceRange.aspectMask = VK_IMAGE_ASPECT_COLOR_BIT;
355  colorAttachmentView.subresourceRange.baseMipLevel = 0;
356  colorAttachmentView.subresourceRange.levelCount = 1;
357  colorAttachmentView.subresourceRange.baseArrayLayer = 0;
358  colorAttachmentView.subresourceRange.layerCount = 1;
359  colorAttachmentView.viewType = VK_IMAGE_VIEW_TYPE_2D;
360  colorAttachmentView.flags = 0;
361 
362  buffers[i].image = images[i];
363 
364  // Transform images from initial (undefined) to present layout
365  vkTools::setImageLayout(
366  cmdBuffer,
367  buffers[i].image,
368  VK_IMAGE_ASPECT_COLOR_BIT,
369  VK_IMAGE_LAYOUT_UNDEFINED,
370  VK_IMAGE_LAYOUT_PRESENT_SRC_KHR);
371 
372  colorAttachmentView.image = buffers[i].image;
373 
374  err = vkCreateImageView(device, &colorAttachmentView, nullptr, &buffers[i].view);
375  assert(!err);
376  }
377  }
