78 STBRP_DEF void stbrp_pack_rects (stbrp_context *context, stbrp_rect *rects, int num_rects);
79 // Assign packed locations to rectangles. The rectangles are of type
80 // 'stbrp_rect' defined below, stored in the array 'rects', and there
81 // are 'num_rects' many of them.
82 //
83 // Rectangles which are successfully packed have the 'was_packed' flag
84 // set to a non-zero value and 'x' and 'y' store the minimum location
85 // on each axis (i.e. bottom-left in cartesian coordinates, top-left
86 // if you imagine y increasing downwards). Rectangles which do not fit
87 // have the 'was_packed' flag set to 0.
88 //
89 // You should not try to access the 'rects' array from another thread
90 // while this function is running, as the function temporarily reorders
91 // the array while it executes.
92 //
93 // To pack into another rectangle, you need to call stbrp_init_target
94 // again. To continue packing into the same rectangle, you can call
95 // this function again. Calling this multiple times with multiple rect
96 // arrays will probably produce worse packing results than calling it
97 // a single time with the full rectangle array, but the option is
98 // available.

115 STBRP_DEF void stbrp_init_target (stbrp_context *context, int width, int height, stbrp_node *nodes, int num_nodes);
116 // Initialize a rectangle packer to:
117 // pack a rectangle that is 'width' by 'height' in dimensions
118 // using temporary storage provided by the array 'nodes', which is 'num_nodes' long
119 //
120 // You must call this function every time you start packing into a new target.
121 //
122 // There is no "shutdown" function. The 'nodes' memory must stay valid for
123 // the following stbrp_pack_rects() call (or calls), but can be freed after
124 // the call (or calls) finish.
125 //
126 // Note: to guarantee best results, either:
127 // 1. make sure 'num_nodes' >= 'width'
128 // or 2. call stbrp_allow_out_of_mem() defined below with 'allow_out_of_mem = 1'
129 //
130 // If you don't do either of the above things, widths will be quantized to multiples
131 // of small integers to guarantee the algorithm doesn't run out of temporary storage.
132 //
133 // If you do #2, then the non-quantized algorithm will be used, but the algorithm
134 // may run out of temporary storage and be unable to pack some rectangles.

218 STBRP_DEF void stbrp_setup_allow_out_of_mem(stbrp_context *context, int allow_out_of_mem)
219 {
220  if (allow_out_of_mem)
221  // if it's ok to run out of memory, then don't bother aligning them;
222  // this gives better packing, but may fail due to OOM (even though
223  // the rectangles easily fit). @TODO a smarter approach would be to only
224  // quantize once we've hit OOM, then we could get rid of this parameter.
225  context->align = 1;
226  else {
227  // if it's not ok to run out of memory, then quantize the widths
228  // so that num_nodes is always enough nodes.
229  //
230  // I.e. num_nodes * align >= width
231  // align >= width / num_nodes
232  // align = ceil(width/num_nodes)
233 
234  context->align = (context->width + context->num_nodes-1) / context->num_nodes;
235  }
236 }

238 STBRP_DEF void stbrp_init_target(stbrp_context *context, int width, int height, stbrp_node *nodes, int num_nodes)
239 {
240  int i;
241 #ifndef STBRP_LARGE_RECTS
242  STBRP_ASSERT(width <= 0xffff && height <= 0xffff);
243 #endif
244 
245  for (i=0; i < num_nodes-1; ++i)
246  nodes[i].next = &nodes[i+1];
247  nodes[i].next = NULL;
248  context->init_mode = STBRP__INIT_skyline;
249  context->heuristic = STBRP_HEURISTIC_Skyline_default;
250  context->free_head = &nodes[0];
251  context->active_head = &context->extra[0];
252  context->width = width;
253  context->height = height;
254  context->num_nodes = num_nodes;
255  stbrp_setup_allow_out_of_mem(context, 0);
256 
257  // node 0 is the full width, node 1 is the sentinel (lets us not store width explicitly)
258  context->extra[0].x = 0;
259  context->extra[0].y = 0;
260  context->extra[0].next = &context->extra[1];
261  context->extra[1].x = (stbrp_coord) width;
262 #ifdef STBRP_LARGE_RECTS
263  context->extra[1].y = (1<<30);
264 #else
265  context->extra[1].y = 65535;
266 #endif
267  context->extra[1].next = NULL;

270 // find minimum y position if it starts at x1
271 static int stbrp__skyline_find_min_y(stbrp_context *c, stbrp_node *first, int x0, int width, int *pwaste)
272 {
273  stbrp_node *node = first;
274  int x1 = x0 + width;
275  int min_y, visited_width, waste_area;
276  STBRP_ASSERT(first->x <= x0);
277 
278  #if 0
279  // skip in case we're past the node
280  while (node->next->x <= x0)
281  ++node;
282  #else
283  STBRP_ASSERT(node->next->x > x0); // we ended up handling this in the caller for efficiency
284  #endif
285 
286  STBRP_ASSERT(node->x <= x0);
287 
288  min_y = 0;
289  waste_area = 0;
290  visited_width = 0;
291  while (node->x < x1) {
292  if (node->y > min_y) {
293  // raise min_y higher.
294  // we've accounted for all waste up to min_y,
295  // but we'll now add more waste for everything we've visted
296  waste_area += visited_width * (node->y - min_y);
297  min_y = node->y;
298  // the first time through, visited_width might be reduced
299  if (node->x < x0)
300  visited_width += node->next->x - x0;
301  else
302  visited_width += node->next->x - node->x;
303  } else {
304  // add waste area
305  int under_width = node->next->x - node->x;
306  if (under_width + visited_width > width)
307  under_width = width - visited_width;
308  waste_area += under_width * (min_y - node->y);
309  visited_width += under_width;
310  }
311  node = node->next;
312  }
313 
314  *pwaste = waste_area;
315  return min_y;
316 }

324 static stbrp__findresult stbrp__skyline_find_best_pos(stbrp_context *c, int width, int height)
325 {
326  int best_waste = (1<<30), best_x, best_y = (1 << 30);
327  stbrp__findresult fr;
328  stbrp_node **prev, *node, *tail, **best = NULL;
329 
330  // align to multiple of c->align
331  width = (width + c->align - 1);
332  width -= width % c->align;
333  STBRP_ASSERT(width % c->align == 0);
334 
335  node = c->active_head;
336  prev = &c->active_head;
337  while (node->x + width <= c->width) {
338  int y,waste;
339  y = stbrp__skyline_find_min_y(c, node, node->x, width, &waste);
340  if (c->heuristic == STBRP_HEURISTIC_Skyline_BL_sortHeight) { // actually just want to test BL
341  // bottom left
342  if (y < best_y) {
343  best_y = y;
344  best = prev;
345  }
346  } else {
347  // best-fit
348  if (y + height <= c->height) {
349  // can only use it if it first vertically
350  if (y < best_y || (y == best_y && waste < best_waste)) {
351  best_y = y;
352  best_waste = waste;
353  best = prev;
354  }
355  }
356  }
357  prev = &node->next;
358  node = node->next;
359  }
360 
361  best_x = (best == NULL) ? 0 : (*best)->x;
362 
363  // if doing best-fit (BF), we also have to try aligning right edge to each node position
364  //
365  // e.g, if fitting
366  //
367  // ____________________
368  // |____________________|
369  //
370  // into
371  //
372  // | |
373  // | ____________|
374  // |____________|
375  //
376  // then right-aligned reduces waste, but bottom-left BL is always chooses left-aligned
377  //
378  // This makes BF take about 2x the time
379 
380  if (c->heuristic == STBRP_HEURISTIC_Skyline_BF_sortHeight) {
381  tail = c->active_head;
382  node = c->active_head;
383  prev = &c->active_head;
384  // find first node that's admissible
385  while (tail->x < width)
386  tail = tail->next;
387  while (tail) {
388  int xpos = tail->x - width;
389  int y,waste;
390  STBRP_ASSERT(xpos >= 0);
391  // find the left position that matches this
392  while (node->next->x <= xpos) {
393  prev = &node->next;
394  node = node->next;
395  }
396  STBRP_ASSERT(node->next->x > xpos && node->x <= xpos);
397  y = stbrp__skyline_find_min_y(c, node, xpos, width, &waste);
398  if (y + height < c->height) {
399  if (y <= best_y) {
400  if (y < best_y || waste < best_waste || (waste==best_waste && xpos < best_x)) {
401  best_x = xpos;
402  STBRP_ASSERT(y <= best_y);
403  best_y = y;
404  best_waste = waste;
405  best = prev;
406  }
407  }
408  }
409  tail = tail->next;
410  }
411  }
412 
413  fr.prev_link = best;
414  fr.x = best_x;
415  fr.y = best_y;
416  return fr;
417 }

419 static stbrp__findresult stbrp__skyline_pack_rectangle(stbrp_context *context, int width, int height)
420 {
421  // find best position according to heuristic
422  stbrp__findresult res = stbrp__skyline_find_best_pos(context, width, height);
423  stbrp_node *node, *cur;
424 
425  // bail if:
426  // 1. it failed
427  // 2. the best node doesn't fit (we don't always check this)
428  // 3. we're out of memory
429  if (res.prev_link == NULL || res.y + height > context->height || context->free_head == NULL) {
430  res.prev_link = NULL;
431  return res;
432  }
433 
434  // on success, create new node
435  node = context->free_head;
436  node->x = (stbrp_coord) res.x;
437  node->y = (stbrp_coord) (res.y + height);
438 
439  context->free_head = node->next;
440 
441  // insert the new node into the right starting point, and
442  // let 'cur' point to the remaining nodes needing to be
443  // stiched back in
444 
445  cur = *res.prev_link;
446  if (cur->x < res.x) {
447  // preserve the existing one, so start testing with the next one
448  stbrp_node *next = cur->next;
449  cur->next = node;
450  cur = next;
451  } else {
452  *res.prev_link = node;
453  }
454 
455  // from here, traverse cur and free the nodes, until we get to one
456  // that shouldn't be freed
457  while (cur->next && cur->next->x <= res.x + width) {
458  stbrp_node *next = cur->next;
459  // move the current node to the free list
460  cur->next = context->free_head;
461  context->free_head = cur;
462  cur = next;
463  }
464 
465  // stitch the list back in
466  node->next = cur;
467 
468  if (cur->x < res.x + width)
469  cur->x = (stbrp_coord) (res.x + width);
470 
471 #ifdef _DEBUG
472  cur = context->active_head;
473  while (cur->x < context->width) {
474  STBRP_ASSERT(cur->x < cur->next->x);
475  cur = cur->next;
476  }
477  STBRP_ASSERT(cur->next == NULL);
478 
479  {
480  stbrp_node *L1 = NULL, *L2 = NULL;
481  int count=0;
482  cur = context->active_head;
483  while (cur) {
484  L1 = cur;
485  cur = cur->next;
486  ++count;
487  }
488  cur = context->free_head;
489  while (cur) {
490  L2 = cur;
491  cur = cur->next;
492  ++count;
493  }
494  STBRP_ASSERT(count == context->num_nodes+2);
495  }
496 #endif
497 
498  return res;
499 }

536 STBRP_DEF void stbrp_pack_rects(stbrp_context *context, stbrp_rect *rects, int num_rects)
537 {
538  int i;
539 
540  // we use the 'was_packed' field internally to allow sorting/unsorting
541  for (i=0; i < num_rects; ++i) {
542  rects[i].was_packed = i;
543  #ifndef STBRP_LARGE_RECTS
544  STBRP_ASSERT(rects[i].w <= 0xffff && rects[i].h <= 0xffff);
545  #endif
546  }
547 
548  // sort according to heuristic
549  STBRP_SORT(rects, num_rects, sizeof(rects[0]), rect_height_compare);
550 
551  for (i=0; i < num_rects; ++i) {
552  if (rects[i].w == 0 || rects[i].h == 0) {
553  rects[i].x = rects[i].y = 0; // empty rect needs no space
554  } else {
555  stbrp__findresult fr = stbrp__skyline_pack_rectangle(context, rects[i].w, rects[i].h);
556  if (fr.prev_link) {
557  rects[i].x = (stbrp_coord) fr.x;
558  rects[i].y = (stbrp_coord) fr.y;
559  } else {
560  rects[i].x = rects[i].y = STBRP__MAXVAL;
561  }
562  }
563  }
564 
565  // unsort
566  STBRP_SORT(rects, num_rects, sizeof(rects[0]), rect_original_order);
567 
568  // set was_packed flags
569  for (i=0; i < num_rects; ++i)
570  rects[i].was_packed = !(rects[i].x == STBRP__MAXVAL && rects[i].y == STBRP__MAXVAL);
571 }
572 #endif
